define i64 @double_it(i64 %x) {
double_it_entry:

  store i64 %x, ptr %x1, align 8
  %0 = load i64, ptr %x1, align 8
  %1 = icmp eq i64 %0, -9223372036854775808
  %2 = or i1 %1, false
  %3 = mul i64 %0, 2
  %4 = select i1 %2, i64 -9223372036854775808, i64 %3
  ret i64 %4
}

declare void @styio_file_close(i64)

define i32 @main() {

fline_exit:                                       ; preds = %fline_hdr

  ret i32 0

runtime_fail:                                     ; preds = %fline_body

  ret i32 0

runtime_ok2:                                      ; preds = %runtime_ok

}
